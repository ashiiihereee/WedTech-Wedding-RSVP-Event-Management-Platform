const https = require('https');

exports.handler = async function(event) {
  if (event.httpMethod !== 'POST') {
    return { statusCode: 405, body: 'Method Not Allowed' };
  }

  const { name, phone, attendance } = JSON.parse(event.body);

  const smsMessage = attendance === 'yes'
    ? `Dear ${name}, we are overjoyed you're joining us for our Shubh Vivah! See you at Fateh Prakash Palace, Udaipur on 25th April. With love, Bride & Groom`
    : `Dear ${name}, we completely understand and will truly miss you. Your blessings mean everything to us. With love, Bride & Groom`;

  const cleanPhone = phone.replace(/\D/g, '').slice(-10);

  const postData = JSON.stringify({
    route:    'q',
    message:  smsMessage,
    language: 'english',
    flash:    0,
    numbers:  cleanPhone,
  });

  return new Promise((resolve) => {
    const options = {
      hostname: 'www.fast2sms.com',
      path:     '/dev/bulkV2',
      method:   'POST',
      headers:  {
        'authorization': 'dwOTJXlnIq9x1W8fQiNZL70ju4mYFgybKpcDUCtv2VHGA3zMSs9UryMZn6HTBKmcJkvpIaN03QCY1bOP',
        'Content-Type':  'application/json',
        'Content-Length': Buffer.byteLength(postData),
      },
    };

    const req = https.request(options, (res) => {
      let data = '';
      res.on('data', chunk => data += chunk);
      res.on('end', () => {
        console.log('Fast2SMS response:', data);
        resolve({
          statusCode: 200,
          body: JSON.stringify({ success: true, response: data }),
        });
      });
    });

    req.on('error', (err) => {
      console.error('Fast2SMS error:', err);
      resolve({
        statusCode: 500,
        body: JSON.stringify({ success: false, error: err.message }),
      });
    });

    req.write(postData);
    req.end();
  });
};
